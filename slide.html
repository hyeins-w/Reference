<!-- Splide JS -->
<script src="https://cdn.jsdelivr.net/npm/@splidejs/splide@4.1.4/dist/js/splide.min.js"></script>
<!-- AutoScroll Extension -->
<script src="https://cdn.jsdelivr.net/npm/@splidejs/splide-extension-auto-scroll@0.5.3/dist/js/splide-extension-auto-scroll.min.js"></script>

const primarySlider = new Splide('.splide2', {
        type : 'loop', 
        perPage  : 1, 
        perMove  : 1,
        gap  : '0.5rem',
        arrows : false, 
        pagination: false, 
        autoplay : true, 
        interval : 3000,
        pauseOnHover: false,
            // ???? splide2 조작은 그대로 허용합니다.
    }).mount(); 

    // 3. secondarySlider (splide3 - 3개씩) 인스턴스 정의 및 마운트
    const secondarySlider = new Splide('.splide3', {
        type : 'loop',
        perPage : 3,
        perMove : 1,
        gap : '0.5rem',
        arrows: false, 
        pagination: false, 
        autoplay : false, 
        interval : 3000,
        pauseOnHover: false,
        focus: 'center', 
        trimSpace: false, 
    }).mount();
        
    // 4. 이벤트 리스너를 사용하여 수동으로 동기화합니다.
    // splide2가 이동할 때 (move 이벤트), splide3를 같은 인덱스로 이동시킵니다.
    primarySlider.on('move', (newIndex) => {
        // secondarySlider를 primarySlider의 현재 인덱스(newIndex)로 이동시킵니다.
        // focus: 'center' 옵션 덕분에 newIndex가 splide3의 가운데로 오게 됩니다.
        secondarySlider.go(newIndex);
    });
        
    // 5. 추가 조치: 자동 재생의 시작도 맞춰줍니다.
    primarySlider.on('autoplay:play', () => {
        secondarySlider.play();
    });
    primarySlider.on('autoplay:pause', () => {
        secondarySlider.pause();
    });
    // primarySlider의 마운트 이후 secondarySlider가 현재 인덱스를 따라가도록 초기 이동
    secondarySlider.go(primarySlider.index);

const splide6 = new Splide('#promo', {
    type: 'loop',
    focus: 'center',
    perPage: 3,
    gap: '1em',                 // 24px -> 1em
    pagination: false,
    arrows: false,
    drag: false,
    snap: false,
    speed: 700,
    autoplay: true,
    interval: 2600,
    pauseOnHover: false,
}).mount();
const root   = splide6.root;
const track6  = root.querySelector('.splide__track');
const slides6 = root.querySelectorAll('.splide__slide');
function updateScales(){
    const rect6    = track6.getBoundingClientRect();
    const centerX = rect6.left + rect6.width / 2;
    const maxDist = rect6.width / 2;
    slides6.forEach((li) => {
        const card = li.querySelector('.card6');
        const r = li.getBoundingClientRect();
        const slideCenter = r.left + r.width / 2;
        const dist  = Math.abs(centerX - slideCenter);
        const t = Math.max(0, 1 - dist / maxDist); // 0~1
        // 스케일: 0.88 ~ 1.06
        const scale = 0.68 + t * (1.2 - 0.88);
        const blurPx = (1 - t) * 1;    // 1px ~ 0px
        const bright = 0.6 + t * 0.4;  // 0.6 ~ 1.0
        const opacity = 0.7 + t * 0.3; // 0.7 ~ 1.0
        card.style.transform = `scale(${scale}) translateY(0.333em)`;
        card.style.filter    = `blur(${blurPx}px) brightness(${bright})`;
        card.style.opacity   = opacity;
    });
    requestAnimationFrame(updateScales);
}
requestAnimationFrame(updateScales);
