document.querySelectorAll('.ripple-btn').forEach(btn=>{
  btn.addEventListener('click',e=>{
    const r=document.createElement('span');
    r.className='ripple';
    const rect=btn.getBoundingClientRect();
    r.style.left=(e.clientX-rect.left)+'px';
    r.style.top=(e.clientY-rect.top)+'px';
    btn.appendChild(r);
    anime({targets:r,scale:[0,12],opacity:[0.7,0],duration:700,easing:'easeOutCubic',complete:()=>r.remove()});
  });
});

1️⃣4️⃣ 단어 단위 Stagger

Performance first, motion second — but both matter.

HTML

<h2 id="wordText">Performance first, motion second — but both matter.</h2>
<button id="wordPlay" class="btn ripple-btn">단어만</button>
<button id="wordCascade" class="btn ripple-btn">단어→글자</button>

JS

document.addEventListener('anime:ready',()=>{
  const el=document.getElementById('wordText');
  const words=el.textContent.split(' ');
  el.textContent='';
  words.forEach(w=>{
    const span=document.createElement('span');
    span.textContent=w+' ';
    span.className='word';
    span.style.display='inline-block';
    el.appendChild(span);
  });
  function wordOnly(){
    anime({targets:'.word',opacity:[0,1],translateY:[16,0],
      easing:'easeOutExpo',duration:600,delay:anime.stagger(80)});
  }
  function cascade(){
    const all=[];
    el.querySelectorAll('.word').forEach((w,i)=>{
      const letters=w.textContent.trim().split('');
      w.textContent='';
      letters.forEach((l,j)=>{
        const s=document.createElement('span');
        s.textContent=l;
        s.style.display='inline-block';
        s.className='letter';
        w.appendChild(s);
        all.push({el:s,delay:i*120+j*12});
      });
    });
    anime({targets:all.map(a=>a.el),
      opacity:[0,1],translateY:[10,0],
      duration:450,easing:'easeOutQuad',
      delay:(t,i)=>all[i].delay});
  }
  wordOnly();
  document.getElementById('wordPlay').addEventListener('click',wordOnly);
  document.getElementById('wordCascade').addEventListener('click',cascade);
});
